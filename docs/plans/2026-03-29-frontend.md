**`src/lib/auth.ts`:**
```ts
const TOKEN_KEY = "ciq_token"
export const setToken = (t: string) => localStorage.setItem(TOKEN_KEY, t)
export const getToken = () => localStorage.getItem(TOKEN_KEY)
export const clearToken = () => localStorage.removeItem(TOKEN_KEY)

export async function login(email: string, password: string): Promise<string> {
  const form = new URLSearchParams({ username: email, password })
  const res = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/auth/login`, {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: form,
  })
  if (!res.ok) throw new Error("Invalid credentials")
  const { access_token } = await res.json()
  setToken(access_token)
  return access_token
}
```

**Acceptance:** Login page renders both columns, form submits, JWT stored, redirects to `/app/dashboard`, error shows inline.

- Forgot password: single-column, email field, animated card swap to confirmation state
- Middleware: redirect `/app/*` to `/login?next=...` if no token, redirect `/login` to `/app/dashboard` if token exists

**Middleware:**
```ts
// src/middleware.ts
import { NextRequest, NextResponse } from "next/server"
export function middleware(req: NextRequest) {
  const token = req.cookies.get("ciq_token")?.value
  // For client-side token: check for cookie set by client OR use a flag
  // Since we use localStorage (client-only), middleware uses a cookie mirror
  const isApp = req.nextUrl.pathname.startsWith("/app")
  const isAuth = req.nextUrl.pathname.startsWith("/login") || req.nextUrl.pathname.startsWith("/register")
  if (isApp && !token) {
    return NextResponse.redirect(new URL(`/login?next=${req.nextUrl.pathname}`, req.url))
  }
  if (isAuth && token) {
    return NextResponse.redirect(new URL("/app/dashboard", req.url))
  }
}
export const config = { matcher: ["/app/:path*", "/login", "/register"] }
```
Note: set a `ciq_token` cookie from client-side login alongside localStorage.

**Acceptance:** Register page with strength bar, forgot-password with card swap animation, middleware redirects work.
